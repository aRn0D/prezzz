namespace Al\Domain;

final class Employee implements EmployeeInterface,
{
	// properties

	private function __construct(Uuid $identifier, string $name, string $position, \DateTimeInterface $hiredAt)
	{
		$this->id = $identifier;
		$this->name = $name;
		$this->position = $position;
		$this->hireAt = $hiredAt
		$this->firedAt = null;
	}

	public static function hire(Uuid $identifier, string $name, string $forPosition, \DateTimeInterface $hiredAt)
	{
		return new self($identifier, $name, $forPosition, $withSalaryScale, $hiredAt);
	}

	public function promote(string $toNewPosition)
	{
		$this->position = $toNewPosition;
	}

	public function fire(\DateTime $firedAt)
	{
		$this->firedAt = $firedAt;
	}
}

namespace Al\Infrastructure\Repository;

final class EmployeeRepository implements EmployeeRepositoryInterface
{
	// Constructor and property

	public function get(Uuid $identifier)
	{
		if (null === $employee = $this->entityManager->find(Employee::class, (string) $identifier->toString())) {
			throw new NotExistingEmployee((string) $identifier->toString());
		}

		return $employee;
	}

	public function add(EmployeeInterface $employee)
	{
		$this->entityManager->persist($employee);
		$this->entityManager->flush();
	}

	public function remove(EmployeeInterface $employee)
	{
		$this->entityManager->remove($employee);
		$this->entityManager->flush();
	}
}

namespace Al\Presenter\Controller;

class EmployeeController extends Controller
{
	public function hireAction(Request $request)
	{
		$form = $this->createForm(EmployeeType::class, new HireEmployee());
		$form->handleRequest($request);

		if ($form->isSubmitted() && $form->isValid()) {
			$employee = $form->getData();

			Employee::hire(Uuid::uuid4(), $employee->getName(), $employee->getPosition());
			$this->get('employee.repository')->save($employee);

			return $this->redirectToRoute('employee_list');
		}

		return $this->render('employee/hire.html.twig', [
			'form' => $form->createView()
		]);
	}
}

namespace Al\Application\Employee;

final class Employee
{
    // Properties

    public function __construct(string $id, string $name, string $position, string $firedAt = null)
    {
        $this->id = $id;
        $this->name = $name;
        $this->position = $position;
        $this->firedAt = $firedAt;
    }

    public function getId(): string
    {
        return $this->id;
    }

    public function getName(): string
    {
        return $this->name;
    }

    public function getPosition(): string
    {
        return $this->position;
    }

	public function getFiredAt(): string
    {
        return $this->firedAt;
    }
}
